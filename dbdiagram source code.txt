//// -- LEVEL 1
//// -- Tables and References

// Creating tables
Table mahasiswa as M {
  npm int [pk, increment] // auto-increment
  nama varchar
  alamat text
  jurusan_id int [ref: > J.id]
  fakultas_id int [ref: > F.id]
  dosen_wali_nip int [ref: > D.nip]
  ip_matkul_id float [ref: < IPM.total]
}


Table dosen as D {
  nip int [pk, increment]
  nama varchar
  gambar__profil_url varchar
  jurusan_id int [ref: > J.id]
}

Table mata_kuliah as MK {
  id int [pk, increment]
  waktu_ambil date
  nama varchar
  SKS int
  ip_matkul int [ref: > IPM.id]
  jurusan_id int [ref: > J.id]
  dosen_nip varchar [ref: > D.nip]
}

Table jurusan as J {
  id int [pk, increment]
  nama varchar
  fakultas_id int [ref: > F.id]
}

Table fakultas as F {
  id int [pk, increment]
  nama varchar
}

Table ip_matkul as IPM {
  id int [pk, increment]
  matkul_id int
  total float
}



Ref: "dosen"."nip" < "dosen"."nama"
